---
title: Nth Request Fails
description: Verify pagination or polling recovery when a specific request fails.
---

import { Tabs, TabItem } from '@astrojs/starlight/components';

Fail only the third matching request to exercise retry and partial-data behavior without making every request fail.

<Tabs>
  <TabItem label="Playwright">

```typescript
await injectChaos(page, {
  seed: 42,
  network: { failures: [{ urlPattern: '/api/search', statusCode: 503, probability: 1, onNth: 3 }] },
});
await page.goto('/search');
for (const term of ['a', 'ab', 'abc']) await page.fill('[name="q"]', term);
await expect(page.locator('[data-testid="retry-inline"]')).toBeVisible();
expect((await getChaosLog(page)).filter((event) => event.type === 'network:failure' && event.applied)).toHaveLength(1);
```

  </TabItem>
  <TabItem label="Cypress">

```typescript
cy.injectChaos({
  seed: 42,

['a', 'ab', 'abc'].forEach((term) => cy.get('[name="q"]').clear().type(term));
```

  <TabItem label="Puppeteer">

```typescript
await injectChaos(page, {
  seed: 42,
  network: { failures: [{ urlPattern: '/api/search', statusCode: 503, probability: 1, onNth: 3 }] },
});
await page.goto('http://localhost:3000/search');
for (const term of ['a', 'ab', 'abc']) {
  await page.click('[name="q"]', { clickCount: 3 });
  await page.type('[name="q"]', term);
}
await page.waitForSelector('[data-testid="retry-inline"]');
const failures = (await getChaosLog(page)).filter((event) => event.type === 'network:failure' && event.applied);
if (failures.length !== 1) throw new Error('Expected one nth failure');
```
